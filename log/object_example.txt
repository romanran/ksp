function createKerbal {
  parameter name.

  function setName {
    parameter newName.
    set name to newName.
  }

  function getName {
    return name.
  }

  return lex(
    "setName", setName@,
    "getName", getName@
  ).
}

local Jeb is createKerbal("Jebediah").
Jeb["setName"]("Bill").
print Jeb["getName"]().
